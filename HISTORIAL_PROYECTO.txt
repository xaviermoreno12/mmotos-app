--- domain/model/Precio.java ---
public record Precio(BigDecimal monto, Moneda moneda) {
    public Precio {
        if (monto.compareTo(BigDecimal.ZERO) < 0)
            throw new IllegalArgumentException("Precio negativo");
    }
    public BigDecimal calcularEnPesos(BigDecimal cotizacion) {
        if (moneda == Moneda.ARS) return monto;
        return monto.multiply(cotizacion).setScale(2, RoundingMode.HALF_UP);
    }
}

--- domain/model/Repuesto.java (abstract) ---
public abstract class Repuesto {
    protected final UUID id;
    protected final String sku;
    protected final String nombre;
    protected Precio precio;
    protected int stockActual;
    protected final int stockMinimo;

    public void validarDisponibilidad(int cantidad) {
        if (stockActual < cantidad)
            throw new InsufficientStockException("Stock insuficiente para " + sku);
    }
    public boolean estaBajoMinimo() { return stockActual <= stockMinimo; }
    void actualizarStock(int delta) { this.stockActual += delta; }
    void actualizarPrecio(Precio nuevoPrecio) { this.precio = nuevoPrecio; }
    public abstract void descontarStock(int cantidad);
}

--- domain/model/KitComposicion.java (Composite Pattern) ---
public class KitComposicion extends Repuesto {
    private final List<ComponenteKit> componentes;
    private BigDecimal precioPromocional;

    @Override
    public void descontarStock(int cantidad) {
        // FASE 1: valida TODOS (atomicidad)
        componentes.forEach(c ->
            c.repuesto().validarDisponibilidad(c.cantidad() * cantidad));
        // FASE 2: descuenta en cascada
        componentes.forEach(c ->
            c.repuesto().descontarStock(c.cantidad() * cantidad));
    }

    public BigDecimal obtenerPrecioFinal(BigDecimal cotizacion) {
        if (precioPromocional != null) return precioPromocional;
        return componentes.stream()
            .map(c -> c.repuesto().precio.calcularEnPesos(cotizacion)
                       .multiply(BigDecimal.valueOf(c.cantidad())))
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }
}

public class Venta {
    private UUID id;
    private List<LineaVenta> lineas;
    private List<Pago> pagos;
    private TipoFactura tipoFactura;
    private EstadoFiscal estadoFiscal;
    private String cae;
    private String numeroTicket;

    public boolean estaBalanceada() {
        BigDecimal totalLineas = lineas.stream()
            .map(l -> l.precioHistorico().multiply(BigDecimal.valueOf(l.cantidad())))
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        BigDecimal totalPagos = pagos.stream()
            .map(Pago::monto).reduce(BigDecimal.ZERO, BigDecimal::add);
        return totalPagos.compareTo(totalLineas) >= 0;
    }
}

public class VentaBuilder {
    private final FiscalPort fiscalPort;
    private List<LineaVenta> lineas = new ArrayList<>();
    private List<Pago> pagos = new ArrayList<>();
    private TipoFactura tipoFactura = TipoFactura.B;

    public VentaBuilder agregarLinea(Repuesto r, int cant, BigDecimal precio) {
        lineas.add(new LineaVenta(r, cant, precio));
        return this;
    }
    public VentaBuilder agregarPago(Pago pago) {
        pagos.add(pago); return this;
    }
    public Venta build() {
        if (tipoFactura != TipoFactura.NO_FISCAL && !fiscalPort.isHardwareReady())
            throw new HardwareFailureException("Impresora fiscal no disponible");
        Venta v = new Venta(UUID.randomUUID(), lineas, pagos, tipoFactura);
        if (!v.estaBalanceada())
            throw new PaymentInconsistencyException("Pago insuficiente");
        return v;
    }
}

public void registrarFallo(String error) {
    this.intentos++;
    this.ultimoError = error;
    this.estado = SyncStatus.FAILED;
    // Backoff: n^2 minutos (1, 4, 9, 16, 25...)
    long minutosEspera = (long) Math.pow(intentos, 2);
    this.proximaSincronizacion = LocalDateTime.now().plusMinutes(minutosEspera);
}
public void marcarSynced() {
    this.estado = SyncStatus.SYNCED;
    this.proximaSincronizacion = null;
}

@Service
public class NoFiscalAdapter implements FiscalPort {
    private final AtomicLong contador = new AtomicLong(0);

    @Override public boolean isHardwareReady() { return true; }

    @Override
    public FiscalResponse emitir(Venta venta) {
        long seq = contador.incrementAndGet();
        String fecha = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd"));
        String ticket = String.format("INT-%s-%05d", fecha, seq);
        return new FiscalResponse(ticket, null, "APROBADO");
    }
}

--- output/fiscal/AfipAdapter.java (fragmento clave) ---
@Override
public FiscalResponse emitir(Venta venta) {
    // Idempotencia: si ya tiene CAE para este ticket, devolverlo
    Optional<String> caeExistente = consultarCaeExistente(venta.getNumeroTicket());
    if (caeExistente.isPresent())
        return new FiscalResponse(venta.getNumeroTicket(), caeExistente.get(), "APROBADO");

    HttpRequest request = HttpRequest.newBuilder()
        .uri(URI.create(props.fiscal().afip().baseUrl() + "/wsfev1/service.asmx"))
        .timeout(Duration.ofMillis(props.fiscal().afip().timeoutMs()))
        .POST(HttpRequest.BodyPublishers.ofString(buildSoapPayload(venta)))
        .header("Content-Type", "text/xml")
        .build();
    try {
        HttpResponse<String> response = httpClient.send(request, BodyHandlers.ofString());
        return parseAfipResponse(response.body());
    } catch (IOException | InterruptedException e) {
        throw new FiscalException("Error comunicando con AFIP: " + e.getMessage());
    }
}

--- output/fiscal/HasarAdapter.java (usa SerialDriver, no JSSC directo) ---
@Override
public FiscalResponse emitir(Venta venta) {
    if (!serialDriver.estaConectado())
        serialDriver.abrir(props.fiscal().hasar().puertoSerie(),
                           props.fiscal().hasar().baudRate());
    byte[] cmd = buildCommand(venta);
    serialDriver.enviar(cmd);
    byte[] resp = serialDriver.recibir(5000);
    return parseHasarResponse(resp);
}
private byte[] buildCommand(Venta venta) {
    ByteArrayOutputStream out = new ByteArrayOutputStream();
    out.write(0x02); // STX
    // datos venta en protocolo Hasar
    out.write(0x03); // ETX
    return out.toByteArray();
}

--- RealizarVentaUseCase.java ---
@Service @RequiredArgsConstructor
public class RealizarVentaUseCase {
    private final VentaTransaccionHelper transaccionHelper;
    private final FiscalPort fiscalPort;
    private final ConfiguracionRepository configuracionRepository;

    public VentaResponse ejecutar(VentaRequest request) {
        BigDecimal cotizacion = configuracionRepository
            .getCotizacionDolar().orElse(BigDecimal.ONE);

        // TX1: validar stock + descontar + guardar PENDIENTE
        Venta venta = transaccionHelper.guardarPendiente(request, cotizacion, fiscalPort);

        // Llamada fiscal FUERA de TX (timeout hasta 10s con AFIP)
        try {
            FiscalPort.FiscalResponse fr = fiscalPort.emitir(venta);
            // TX2: confirmar con CAE
            return transaccionHelper.confirmarFiscal(venta.getId(), fr);
        } catch (FiscalException e) {
            // TX3: marcar ERROR_HARDWARE
            transaccionHelper.marcarError(venta.getId());
            throw e;
        }
    }
}

--- VentaTransaccionHelper.java (FIX: self-invocation @Transactional) ---
// PROBLEMA ORIGINAL: Spring AOP no intercepta llamadas 'this.metodo()'
// La anotacion @Transactional NO se aplica cuando se llama desde el mismo bean
// SOLUCION: extraer a @Service separado para que Spring inyecte el proxy correcto

@Service @RequiredArgsConstructor
public class VentaTransaccionHelper {

    @Transactional
    public Venta guardarPendiente(VentaRequest request, BigDecimal cotizacion,
                                   FiscalPort fiscalPort) {
        // Lock pesimista por producto para evitar race conditions
        request.lineas().forEach(l -> {
            ProductoEntity pe = productoRepo.findByIdWithLock(l.productoId()).orElseThrow();
            if (pe.getStockActual() < l.cantidad())
                throw new InsufficientStockException("Stock insuficiente: " + pe.getSku());
            pe.setStockActual(pe.getStockActual() - l.cantidad());
            productoRepo.save(pe);
        });
        VentaEntity ve = ventaMapper.toVentaEntity(request, cotizacion);
        ventaRepo.save(ve);
        syncLogRepo.save(new SyncLogEntity(UUID.randomUUID(), ve.getId(), "VENTA"));
        return ventaMapper.toDomain(ve);
    }

    @Transactional
    public VentaResponse confirmarFiscal(UUID ventaId, FiscalPort.FiscalResponse fr) {
        VentaEntity v = ventaRepo.findById(ventaId).orElseThrow();
        v.setNumeroTicket(fr.numeroTicket());
        v.setCae(fr.cae());
        v.setEstadoFiscal(EstadoFiscal.APROBADO);
        v.setSyncStatus(SyncStatus.PENDING);
        ventaRepo.save(v);
        return new VentaResponse(v.getId(), v.getNumeroTicket(), v.getCae(),
            "APROBADO", "PENDING", v.getTotal(), v.getCreatedAt());
    }

    @Transactional
    public void marcarError(UUID ventaId) {
        ventaRepo.findById(ventaId).ifPresent(v -> {
            v.setEstadoFiscal(EstadoFiscal.ERROR_HARDWARE);
            ventaRepo.save(v);
        });
    }
}

@Service @RequiredArgsConstructor
public class SyncVentaProcessor {
    private final SyncLogJpaRepository syncLogRepo;
    private final N8nAdapter n8nAdapter;

    @Transactional
    public void procesar(SyncLogEntity log) {
        try {
            n8nAdapter.enviarAlertaSync(log.getEntidadId(), log.getTipoEntidad());
            log.marcarSynced();

            log.registrarFallo(e.getMessage());

            syncLogRepo.save(log);
        }
    }
}

    public Optional<ProductoDTO> porId(UUID id) {
        return repo.findById(id).map(ProductoMapper::toDTO);
    }

    public List<ProductoDTO> buscar(String termino) {
        List<ProductoEntity> resultado = new ArrayList<>();
        repo.findBySkuAndActivoTrue(termino).ifPresent(resultado::add);
        resultado.addAll(repo.buscarPorNombreSimilar(termino));
        return resultado.stream().distinct().map(ProductoMapper::toDTO).toList();
    }
}

        return new LoginResponse(token, usuario.getUsername(), usuario.getRol(),
                                 usuario.getId().toString());
    }

    public record LoginResponse(String token, String username, String rol, String usuarioId) {}
}

--- config/JwtService.java ---
@Service
public class JwtService {
    private final SecretKey key;
    private final long expirationMs;

    public JwtService(AppProperties props) {
        this.key = Keys.hmacShaKeyFor(
            props.jwt().secret().getBytes(StandardCharsets.UTF_8));
        this.expirationMs = props.jwt().expirationMs();
    }

    public String generarToken(String username, String rol) {
        return Jwts.builder()
            .subject(username)
            .claim("rol", rol)
            .issuedAt(new Date())
            .expiration(new Date(System.currentTimeMillis() + expirationMs))
            .signWith(key).compact();
    }

    public boolean esValido(String token) {
        try { parsear(token); return true; }
        catch (JwtException | IllegalArgumentException e) { return false; }
    }

    public String extractUsername(String token) {
        return parsear(token).getPayload().getSubject();
    }

    private Jws<Claims> parsear(String token) {
        return Jwts.parser().verifyWith(key).build().parseSignedClaims(token);
    }
}

--- serve.py ---
class SPAHandler(http.server.SimpleHTTPRequestHandler):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, directory=str(DIST_DIR), **kwargs)

    def do_GET(self):
        if self.path.startswith("/api"):
            self._proxy()
        else:
            file_path = DIST_DIR / self.path.lstrip("/").split("?")[0]
            if not file_path.exists() or file_path.is_dir():
                self.path = "/index.html"     # SPA routing
            super().do_GET()

    def do_POST(self):
        if self.path.startswith("/api"): self._proxy()
        else: self.send_error(405)

    def do_PUT(self):
        if self.path.startswith("/api"): self._proxy()
        else: self.send_error(405)

    def _proxy(self):
        target = "http://localhost:8080" + self.path
        length = int(self.headers.get("Content-Length", 0))
        body = self.rfile.read(length) if length > 0 else None
        headers = {k: v for k, v in self.headers.items()
                   if k.lower() not in ("host", "content-length")}
        try:
            req = urllib.request.Request(target, data=body,
                                         method=self.command, headers=headers)
            with urllib.request.urlopen(req, timeout=15) as resp:
                self._send_proxy_response(resp.status, resp.headers, resp.read())
        except urllib.error.HTTPError as e:
            self._send_proxy_response(e.code, e.headers, e.read())
        except Exception as exc:
            self.send_error(502, f"Backend no disponible: {exc}")

BUG 1: @Transactional self-invocation (Spring AOP proxy bypass)
  Problema: spring solo intercepta llamadas a traves del proxy.

  Solucion: SyncVentaProcessor @Service separado

BUG 3: @OneToMany(mappedBy="ventaId") sobre campo UUID
  Problema: JPA espera que mappedBy referencie un campo @ManyToOne del tipo padre
            No funciona sobre un UUID puro -> JPA AnnotationException en startup
  Solucion: Eliminar @OneToMany, VentaDetalleJpaRepository + PagoJpaRepository
            separados, persistir hijos individualmente en VentaRepositoryAdapter

BUG 4: BCrypt lanza exception con hash placeholder
  Problema: '$2a$12$placeholder...' no es BCrypt valido (longitud incorrecta)
            BCryptPasswordEncoder.matches() tira IllegalArgumentException -> HTTP 500
  Solucion: V2__seed_usuarios.sql con hash real:
            python -c "import bcrypt; print(bcrypt.hashpw(b'admin123', bcrypt.gensalt(10)).decode())"
            -> '$2b$10$5NjtTt0jdAPGb.j1ZofeQOWNvsfHlgZ6FWR99F.XyEArjT8SPciPW'

  Solucion: toDetalleEntities() y toPagoEntities() retornan List nuevas

  Solucion: Agregar PagoMercadoPago record + actualizar 'permits'
